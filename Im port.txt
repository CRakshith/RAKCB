services:
  # 1. DEVELOPMENT SERVICE
  web-dev:
    build: ./web-dev               # Short syntax! Automatically finds './web-dev/Dockerfile'
    image: my-app:dev
    container_name: rcb-app-dev
    ports:
      - "8090:80"

  # 2. PRODUCTION SERVICE
  web-prod:
    build: ./web-prod              # Short syntax! Automatically finds './web-prod/Dockerfile'
    image: my-app:1.2.0
    container_name: rcb-app-prod
    ports:
      - "80:80"
	  
  # 3. DATABASE (Pulled directly from Docker Hub)
  my-database:                     # The service name
    image: postgres:15-alpine      # No build step! Downloads directly from Docker Hub
    container_name: rcb-db-container
    environment:                   # Sets internal settings required by the Hub image
      - POSTGRES_USER=admin
      - POSTGRES_PASSWORD=secret_password_123
      - POSTGRES_DB=mydb

	  Docker swarm:
	  
	  Initialize master node:
	  
	  * docker swarm init --advertise-addr <private ip of EC2>
	  
	 * join node to the manager or join node as manager: 
	  docker swarm join-token [OPTIONS] (worker|manager)
	  By using: 
	   docker swarm join-token worker
	   you will get the command to join the worker to this master
	  By using: 
	   docker swarm join-token manager
	   you will get the command to join the manager(master) to this master
	   
	   
	  * To check the nodes in the swarm:
	 => docker node ls
	   
	   **
	   How to run container using swarm
	 => docker service create --name <Service name> -p 8080:80 <image name>
	 
	  how to check the number of containers and it's replica in master:
	 => docker service ls
	  
	  how to remove the service:
	 => docker service rm <service name/id>
	 
	  how to scale the service:
	  => docker service scale <Service name>=<number of replica required>
	  
	  how to update the service?
	 => docker service update --image <latest image name> <Service Name>
